        # setup model layers
        model = tf.keras.models.Sequential()

        model.add(tf.keras.layers.Dropout(0.5))
        model.add(tf.keras.layers.Dense(2, activation='softmax'))
        model.compile(loss=self.loss, optimizer=self.optimizer)
        self.model = model
